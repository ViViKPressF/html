    1. Базовые стили текста 

    /* Основные стили текста */
body {
  font-family: Arial, sans-serif;
  font-size: 16px;
  line-height: 1.5;
  color: #333;
}

h1 {
  font-size: 2em;
  color: #0066cc;
  margin-bottom: 20px;
}

p {
  margin-bottom: 15px;
}

    2. Стилизация ссылок

    /* Стили для ссылок */
a {
  color: #0066cc;
  text-decoration: none;
}

a:hover {
  text-decoration: underline;
  color: #004499;
}

    3. Стили для кнопок

    /* Стили кнопок */
.button {
  display: inline-block;
  padding: 10px 20px;
  background-color: #0066cc;
  color: white;
  border-radius: 4px;
  text-align: center;
  cursor: pointer;
}

.button:hover {
  background-color: #004499;
}

    4. Flexbox раскладка

    /* Flexbox контейнер */
.container {
  display: flex;
  justify-content: space-between;
  align-items: center;
  gap: 20px;
}

    5. Grid раскладка

    /* Grid контейнер */
.grid {
  display: grid;
  grid-template-columns: repeat(3, 1fr);
  gap: 15px;
}

    6. Адаптивные медиа-запросы

    /* Медиа-запросы для адаптива */
@media (max-width: 768px) {
  .container {
    flex-direction: column;
  }
  
  .grid {
    grid-template-columns: 1fr;
  }
}

    7. Стили для форм

    /* Стилизация форм */
input, textarea {
  padding: 8px 12px;
  border: 1px solid #ddd;
  border-radius: 4px;
  width: 100%;
  margin-bottom: 10px;
}

input:focus, textarea:focus {
  outline: none;
  border-color: #0066cc;
}

    8. Анимации

    /* Простая анимация */
@keyframes fadeIn {
  from { opacity: 0; }
  to { opacity: 1; }
}

.animated {
  animation: fadeIn 0.5s ease-in-out;
}

    9. Стили для таблиц

    /* Стилизация таблиц */
table {
  width: 100%;
  border-collapse: collapse;
}

th, td {
  padding: 12px;
  text-align: left;
  border-bottom: 1px solid #ddd;
}

th {
  background-color: #f5f5f5;
}

    10. Общие утилитные классы

    /* Утилитные классы */
.text-center { text-align: center; }
.mt-20 { margin-top: 20px; }
.p-15 { padding: 15px; }
.hidden { display: none; }
